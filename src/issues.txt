static char	*read_line_no_buffer(int fd)
{
	char	buffer[1];
	char	*line;
	char	*tmp;
	int		ret;

	line = ft_strdup("");
	if (!line)
		return (NULL);
	while (1)
	{
		ret = read(fd, buffer, 1);
		if (ret <= 0)
		{
			if (ft_strlen(line) == 0)
				return (free(line), NULL);
			return (line);
		}
		if (buffer[0] == '\n')
			return (line);
		tmp = line;
		line = ft_strjoin_char(tmp, buffer[0]);
		free(tmp);
		if (! line)
			return (NULL);
	}
}

	int		pipefd[2];
	pid_t	pid;

	if (pipe(pipefd) == -1)
		return (-1);

	if (pid == 0)

		signal(SIGQUIT, SIG_IGN);
